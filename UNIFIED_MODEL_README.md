## Quick Start

```python
from unified_model_loader import load_model, load_sovereign

# Load the sovereign (maximum) model
model = load_sovereign()

# Check if model contains a junction
if model.contains(0.5):
    print("Junction found!")

# Find nearest junction to a value
nearest, idx = model.find_nearest(0.12345)
print(f"Nearest: {nearest} at index {idx}")

# Get all junctions in a range
junctions = model.get_range(-0.5, 0.5)
print(f"Found {len(junctions)} junctions in range")
```
